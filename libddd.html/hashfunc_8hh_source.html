41  inline size_t
42  wang32_hash(size_t key)
43  {
44  // We assume that size_t has at least 32bits.
45  key += ~(key << 15);
46  key ^= (key >> 10);
47  key += (key << 3);
48  key ^= (key >> 6);
49  key += ~(key << 11);
50  key ^= (key >> 16);
51  return key;
52  }

72  inline size_t
73  knuth32_hash(size_t key)
74  {
75  // 2654435761 is the golden ratio of 2^32. The right shift of 3
